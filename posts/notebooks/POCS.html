def pocs(dataR, calib):
    idx = np.where(np.sum(np.abs(dataR), axis=1)==0)[0]
    phs = np.exp(1j*np.angle(ifft2c(calib)))
    im = np.zeros(dataR.shape, dtype = complex)
    for n in range(5):
        tmp = im*phs
        tmp = fft2c(tmp)
        tmp[:idx[0],:] = dataR[:idx[0],:] 
        tmp = ifft2c(tmp)*np.conj(phs)
        tmp[tmp < 0] = 0 
        im = tmp 
    return im
